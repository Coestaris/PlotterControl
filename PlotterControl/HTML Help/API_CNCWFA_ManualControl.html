Библиотека CWA_ManualControl

Библиотека CWA_MathParser - специфическая библиотека для ручного управления устройством. Состоит из:

  • Класс ManualControl.
Класс ManualControl

Имеет конструктор:

  • ManualControl(string pname, int bd), где pname и bd - параметры порта.

Имеет следующие методы:

  • public void ToolUp() - поднять перо.
  • public void Close() - закрыть порт.
  • public void ReOpen() - открыть порт.
  • public void ToolDown() - опустить перо.
  • public void MakeStep(System.ConsoleKeyInfo k) - сделать шаг.

Поле Step означает сколько шагов сделает метод MakeStep.

void ManualControl()
{
   ManualControl a;
   try
   {
      args = args[0].Split(',');
      string port = args[0];
      int bd = int.Parse(args[1]);
      int step = int.Parse(args[2]);
      Console.CursorVisible = false;
      a = new ManualControl(port,  bd);
      a.Step = step;
      Console.Clear();
      Console.WriteLine("Press ESC to exit");
   }
     catch
   {
      MessageBox.Show("Launch with wrong arguments","Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
      return;
   }
   while (true)
   {
      var c = Console.ReadKey();
      if (c.Key == ConsoleKey.Escape) Environment.Exit(0);
      a.MakeStep(c);
      Console.Clear();
      Console.WriteLine("Press ESC to exit");
   }
}
